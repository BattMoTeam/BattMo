Composite electrode json input file

Here is json input for the composite electrode part, in Composite Silicon Graphite electrode

{
  "NegativeElectrode": {
    "Coating" : {
      "activeMaterialModelSetup" : {"composite": true},
      "ActiveMaterial1" : {
        "specificHeatCapacity": 632.0,
        "heatCapacity": 632000.0,
        "thermalConductivity": 1.04,
        "electronicConductivity": 100,
        "massFraction" : 0.92,
        "density": 2240,
        "Interface": {
          "isFile": true,
          "filename": "ParameterData/MaterialProperties/Graphite/graphite.json"
        },
        "diffusionModelType" : "full",
        "SolidDiffusion" : {
          "activationEnergyOfDiffusion": 5000,
          "referenceDiffusionCoefficient": 3.9e-14,
          "particleRadius": 1e-06,
          "N": 10
        }
      },
      "ActiveMaterial2" : {
        "specificHeatCapacity": 632.0,
        "heatCapacity": 632000.0,
        "thermalConductivity": 1.04,
        "electronicConductivity": 100,
        "massFraction" : 0.08,
        "density": 2330,        
        "Interface": {
          "isFile": true,
          "filename": "ParameterData/MaterialProperties/Silicon/silicon.json"
        },
        "diffusionModelType" : "full",
        "SolidDiffusion" : {
          "activationEnergyOfDiffusion": 5000,
          "referenceDiffusionCoefficient": 1e-15,
          "particleRadius": 6e-8,
          "N": 10
        }
      }
    }
  }
}
